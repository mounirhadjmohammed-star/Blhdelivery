const CACHE = "blh-cache-v1"; const ASSETS = ["./","index.html","app.js","manifest.json"]; self.addEventListener("install", e => { e.waitUntil(caches.open(CACHE).then(c => c.addAll(ASSETS))); }); self.addEventListener("activate", e => { e.waitUntil( caches.keys().then(keys => Promise.all(keys.filter(k=>k!==CACHE).map(k=>caches.delete(k)))) ); }); self.addEventListener("fetch", e => { e.respondWith( caches.match(e.request).then(res => res || fetch(e.request).catch(()=>caches.match("index.html"))) ); });
